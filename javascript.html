连接字符串

//concat()
        console.log('y'.concat('ooo', 'oo', 'o')); //// "yoooooo"，传参数量不限

获得字符串中的某一个字符

//charAt()
        'yo'.charAt(0); // "y" 程序员是从0开始数的
        'yo'.charAt(1); // "o"
        'yo'[0]; // "y"

检查一段字符是否包含另一段字符

//includes('')
        '花花你好'.includes('花花'); // true
        '花花你好'.includes('拴蛋'); // false

截取字符串

//slice()
        var str = '王花花和小熊跳舞跳呀跳呀一二一';
        str.slice(3, 8); // "和小熊跳舞"，第一个传参为开始索引，第二个传参为结束索引
        str.slice(3); // "和小熊跳舞跳呀跳呀一二一"，若省略第二个传参将会截取至最后一个字符

延时触发

//setTimeout
            var count = 0;

              console.log("count = " + count);

间隔触发

//setInterval
            var count = 0;
            var r = setInterval(function(){
              count++;
              console.log("count = " + count);
            },1000)

清除间隔触发

//clearInterval
            var count = 0;
            var r = setInterval(function(){
              count++;
              if (count > 5){
                clearInterval(r);
                return;
              }
              console.log("count = " + count);
            },1000)
